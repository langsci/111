\begin{document}
\begin{tikzpicture}[->,>=stealth',shorten >=1pt,auto,node distance=3cm,
  thick,main node/.style={circle,draw,font=\large\bfseries}]

  \node[main node] (1) {spa};
  \node[main node] (2) [below of=1] {eng};
  \node[main node] (3) [below left of=2] {kor};
  \node[main node] (4) [below right of=2] {rus};

  \path[<->, every node/.style={font=\sffamily\small}]
    (2) edge node {} (1)
	edge node {} (3)
	edge node {} (4);


\end{tikzpicture}
